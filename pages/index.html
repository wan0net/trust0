Quick Start

git clone https://github.com/wan0net/trust0.git
cd trust0
pnpm install
pnpm --filter @trust0/identity test    # 88 tests
pnpm --filter @trust0/api dev          # API on :8788
pnpm --filter @trust0/web dev          # Web on :5173
